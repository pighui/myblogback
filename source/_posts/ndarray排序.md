```python
import numpy as np
```

```python
# 创建十位员工的基本工资、绩效和奖金
salary = np.random.randint(10000, 30000, size = (10,1))
performance = np.random.randint(1000, 3000, size = (10,1))
bonus = np.random.randint(500, 1500, size = (10,1))

persons = np.hstack((np.hstack((salary,performance)),bonus))
persons
```
